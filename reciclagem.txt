        // --- Firebase Configuration ---

        async function initFirebase() {
            try {
                const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
                appId = typeof __app_id !== 'undefined' ? __app_id : 'a3p-recicla-game';
                
                if (firebaseConfig) {
                    const app = initializeApp(firebaseConfig);
                    auth = getAuth(app);
                    db = getFirestore(app);

                    if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                        await signInWithCustomToken(auth, __initial_auth_token);
                    } else {
                        await signInAnonymously(auth);
                    }

                    onAuthStateChanged(auth, (u) => {
                        user = u;
                        isFirebaseActive = !!u;
                    });
                }
            } catch (error) {
                console.warn("Firebase não configurado ou indisponível. Rodando modo local.", error);
            }
        }

        initFirebase();

        // --- Audio System (Web Audio API) ---
        let audioCtx;

        let audioInitialized = false;
        
        let bgmInterval;

        const tracks = {
            menu: { melody: [261.63, 329.63, 392.00, 523.25, 392.00, 329.63], speed: 300, type: 'triangle', humFreq: 150, playOfficeSounds: true }, 
            countdown: { melody: [], speed: 1000, type: 'sine', humFreq: 150, playOfficeSounds: false }, // Silêncio musical durante a contagem

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
            if (audioCtx.state === 'suspended') {
                audioCtx.resume();
            }
        }

        function playTone(freq, type, duration, vol = 0.1) {
            if (isMuted || !audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = type;
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            
            gain.gain.setValueAtTime(0, audioCtx.currentTime);
            gain.gain.linearRampToValueAtTime(vol, audioCtx.currentTime + 0.05);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + duration);
        }

        function playMusicNote(freq, type, duration, vol = 0.1) {
            if (isMuted || !audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            
            osc.type = type;
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            
            gain.gain.setValueAtTime(0, audioCtx.currentTime);
            gain.gain.linearRampToValueAtTime(vol, audioCtx.currentTime + (duration * 0.1));
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + duration);
        }

        function playOfficeClick() {
            if (isMuted || !audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            const filter = audioCtx.createBiquadFilter();
            
            osc.type = 'square';
            osc.frequency.setValueAtTime(200 + Math.random()*200, audioCtx.currentTime);
            
            filter.type = 'highpass';
            filter.frequency.value = 800;

            gain.gain.setValueAtTime(0.01, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.05);

            osc.connect(filter);
            filter.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.05);
        }

        function startOfficeSounds() {
            if (officeSoundsTimeout) clearTimeout(officeSoundsTimeout);
            
            function scheduleNextClick() {
                if (gameState === 'playing' || gameState === 'menu') {
                    if (!isMuted) playOfficeClick();
                    officeSoundsTimeout = setTimeout(scheduleNextClick, Math.random() * 1800 + 200);
                }
            }
            scheduleNextClick();
        }

        function stopOfficeSounds() {
            if (officeSoundsTimeout) clearTimeout(officeSoundsTimeout);
        }

        function updateAudioState() {
            if (!audioCtx) return;
            
            if (bgmInterval) clearInterval(bgmInterval);
            if (noiseSource) {
                try { noiseSource.stop(); } catch(e){}
                noiseSource.disconnect();
                noiseSource = null;
            }

            if (isMuted) {
                stopOfficeSounds();
                return;
            }

            const track = tracks[gameState] || tracks['menu'];

            // Ambient Hum (Office AC)
            const bufferSize = audioCtx.sampleRate * 2;
            const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
            const output = buffer.getChannelData(0);
            for (let i = 0; i < bufferSize; i++) {
                output[i] = Math.random() * 2 - 1; // White noise
            }
            noiseSource = audioCtx.createBufferSource();
            noiseSource.buffer = buffer;
            noiseSource.loop = true;
            
            noiseFilter = audioCtx.createBiquadFilter();
            noiseFilter.type = 'lowpass';
            noiseFilter.frequency.value = track.humFreq;
            
            noiseGain = audioCtx.createGain();
            noiseGain.gain.value = 0.015; // Subtle AC Volume
            
            noiseSource.connect(noiseFilter);
            noiseFilter.connect(noiseGain);
            noiseGain.connect(audioCtx.destination);
            noiseSource.start();

            // Office Keyboards/Clicks
            if (track.playOfficeSounds) {
                startOfficeSounds();
            } else {
                stopOfficeSounds();
            }

            // Start BGM
            noteIndex = 0;
            if (track.melody.length > 0) {
                bgmInterval = setInterval(() => {
                    if (isMuted || !audioCtx) return;
                    
                    // Ajustamos o volume para ser um pouco mais baixo nas ondas 'square' para não ser estridente
                    const volume = track.type === 'square' ? 0.04 : 0.08;
                    playMusicNote(track.melody[noteIndex], track.type, track.speed / 1000 * 1.5, volume);
                    
                    noteIndex = (noteIndex + 1) % track.melody.length;
                }, track.speed);
            }
        }

        // Sintetizador de Torcida (White Noise Filtered)
        function playCrowdCheer() {
            if (isMuted || !audioCtx) return;
            const bufferSize = audioCtx.sampleRate * 4; // 4 segundos
            const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
            const output = buffer.getChannelData(0);
            for (let i = 0; i < bufferSize; i++) {
                output[i] = Math.random() * 2 - 1;
            }
            const whiteNoise = audioCtx.createBufferSource();
            whiteNoise.buffer = buffer;

            const bandpass = audioCtx.createBiquadFilter();
            bandpass.type = 'bandpass';
            bandpass.frequency.value = 800; // Frequência que lembra aplausos/vozes
            bandpass.Q.value = 0.5;

            const gainNode = audioCtx.createGain();
            gainNode.gain.setValueAtTime(0, audioCtx.currentTime);
            gainNode.gain.linearRampToValueAtTime(0.4, audioCtx.currentTime + 1.5); // Aumenta o volume da torcida
            gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 4); // Vai sumindo

            whiteNoise.connect(bandpass);
            bandpass.connect(gainNode);
            gainNode.connect(audioCtx.destination);

            whiteNoise.start();
        }

        function playCountdownBeep(isFinal) {

        }
        window.addEventListener('resize', resize);
        resize();

        // Input Handling
        function setPlayerTarget(clientX) {
            player.targetX = clientX;
            if (player.targetX < player.width/2) player.targetX = player.width/2;
            if (player.targetX > width - player.width/2) player.targetX = width - player.width/2;
        }

        window.addEventListener('mousemove', (e) => {
            if (gameState === 'playing') setPlayerTarget(e.clientX);
        });

        window.addEventListener('touchmove', (e) => {
            if (gameState === 'playing') {
                e.preventDefault(); 
                setPlayerTarget(e.touches[0].clientX);
            }
        }, { passive: false });

        window.addEventListener('touchstart', (e) => {
             if (gameState === 'playing') setPlayerTarget(e.touches[0].clientX);
        });

        window.addEventListener('keydown', (e) => {
            if (gameState !== 'playing') return;
            const speed = 40;
            if (e.key === 'ArrowLeft') setPlayerTarget(player.targetX - speed);
            if (e.key === 'ArrowRight') setPlayerTarget(player.targetX + speed);
        });

        function spawnItem() {
            const isOrganic = Math.random() < 0.15;
            const filteredItems = itemTypes.filter(i => isOrganic ? i.type === 'bad' : i.type === 'good');
            const template = filteredItems[Math.floor(Math.random() * filteredItems.length)];
            
            const size = 65; 
            items.push({
                x: Math.random() * (width - size*2) + size,
                y: -size,
                size: size,
                speed: 3 + Math.random() * 4 + (60 - timeLeft)*0.05,
                rotation: 0,
                rotSpeed: (Math.random() - 0.5) * 0.1,
                ...template
            });
        }

        function createParticles(x, y, color) {
            for (let i = 0; i < 8; i++) {
                particles.push({
                    x: x,
                    y: y,
                    vx: (Math.random() - 0.5) * 10,
                    vy: (Math.random() - 0.5) * 10,
                    life: 1,
                    color: color
                });
            }
        }

        // Main Loop
        function update(deltaTime) {
            if (gameState !== 'playing') return;

            secondTimer += deltaTime;
            if (secondTimer >= 1000) {
                timeLeft--;
                document.getElementById('timeDisplay').innerText = timeLeft;
                secondTimer = 0;
                
                if (timeLeft <= 0) {
                    endGame();
                }
            }

            player.x += (player.targetX - player.x) * 0.2;

            const spawnChance = 0.02 + ((60 - timeLeft) * 0.0005);
            if (Math.random() < spawnChance) {
                spawnItem();
            }

            for (let i = items.length - 1; i >= 0; i--) {
                const item = items[i];
                item.y += item.speed;
                item.rotation += item.rotSpeed;

                const px = player.x - player.width / 2;
                const py = player.y;
                
                if (item.y + item.size/2 > py && item.y - item.size/2 < py + player.height) {
                    if (item.x + item.size/2 > px && item.x - item.size/2 < px + player.width) {
                        
                        score += item.points;
                        document.getElementById('scoreDisplay').innerText = score;
                        createParticles(item.x, item.y, item.color);
                        
                        if (item.type === 'good') {
                            playCatchSound();
                            
                            // Atualiza os contadores por tipo de material
                            if (item.material) {
                                counters[item.material]++;
                                // Capitaliza a primeira letra para encontrar o ID correto (ex: countPapel)
                                const elementId = 'count' + item.material.charAt(0).toUpperCase() + item.material.slice(1);
                                document.getElementById(elementId).innerText = counters[item.material];
                            }
                            
                            if (score > 0 && Math.floor(score / 50) > lastMilestone) {
                                lastMilestone = Math.floor(score / 50);
                                floatingMessage = {
                                    text: motivationalMessages[Math.floor(Math.random() * motivationalMessages.length)],
                                    alpha: 1.5,
                                    y: height / 2 + 50
                                };
                            }
                        } else {
                            playBadSound();
                            gameState = 'paused';
                            updateAudioState();
                            
                            // Sorteia uma mensagem educativa aleatória
                            const randomMsg = organicMessages[Math.floor(Math.random() * organicMessages.length)];
                            document.getElementById('organicPauseMessage').innerText = randomMsg;
                            
                            document.getElementById('organicPauseScreen').classList.remove('hidden');
                            document.body.style.backgroundColor = '#78350f';
                            document.body.style.backgroundImage = 'none';
                        }
                        
                        items.splice(i, 1);
                        continue;
                    }
                }

                if (item.y > height + item.size) {
                    items.splice(i, 1);
                }
            }

            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.x += p.vx;
                p.y += p.vy;
                p.life -= 0.02;
                if (p.life <= 0) particles.splice(i, 1);
            }
        }

        function draw() {
            ctx.clearRect(0, 0, width, height);

            // Permite desenhar o cenário e o jogador mesmo durante a contagem
            if (gameState !== 'playing' && gameState !== 'paused' && gameState !== 'countdown') return;

            ctx.save();
            ctx.translate(player.x, player.y);
            
            ctx.fillStyle = '#22c55e';
            ctx.beginPath();
            ctx.roundRect(-player.width/2, 0, player.width, player.height, 10);
            ctx.fill();
            
            ctx.fillStyle = '#16a34a';
            ctx.beginPath();
            ctx.roundRect(-player.width/2 - 5, -10, player.width + 10, 15, 5);
            ctx.fill();

            ctx.fillStyle = '#ffffff';
            ctx.font = '36px Arial';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText('♻️', 0, player.height/2);

            ctx.restore();

            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            for (const item of items) {
                ctx.save();
                ctx.translate(item.x, item.y);
                ctx.rotate(item.rotation);
                
                ctx.fillStyle = 'rgba(255,255,255,0.7)';
                ctx.beginPath();
                ctx.arc(0,0, item.size/1.5, 0, Math.PI*2);
                ctx.fill();
                
                ctx.font = `${item.size}px Arial`;
                ctx.fillText(item.emoji, 0, 0);
                ctx.restore();
            }

            for (const p of particles) {
                ctx.save();
                ctx.globalAlpha = p.life;
                ctx.fillStyle = p.color;
                ctx.beginPath();
                ctx.arc(p.x, p.y, 4, 0, Math.PI * 2);
                ctx.fill();
                ctx.restore();
            }

            if (floatingMessage.alpha > 0) {
                ctx.save();
                ctx.globalAlpha = Math.min(1, floatingMessage.alpha);
                ctx.fillStyle = '#16a34a';
                ctx.strokeStyle = '#ffffff';
                ctx.lineWidth = 6;
                ctx.font = '900 48px "Fredoka", sans-serif';
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                
                ctx.strokeText(floatingMessage.text, width / 2, floatingMessage.y);
                ctx.fillText(floatingMessage.text, width / 2, floatingMessage.y);
                ctx.restore();

                floatingMessage.y -= 1.5;
                floatingMessage.alpha -= 0.015;
            }
        }

        function gameLoop(timestamp) {
            const deltaTime = timestamp - lastTime;
            lastTime = timestamp;

            update(deltaTime);
            draw();

            animationId = requestAnimationFrame(gameLoop);
        }

        // --- Game Flow Control ---

        function startGame(e) {
            e.preventDefault();
            const emailInput = document.getElementById('userEmail').value;
            if (!emailInput) return;
            playerEmail = emailInput;

            if (!audioInitialized) {
                initAudio();
                audioInitialized = true;
            }
            
            score = 0;
            timeLeft = 60;
            lastMilestone = 0;
            floatingMessage = { text: '', alpha: 0, y: 0 };
            items.length = 0;
            particles.length = 0;
            
            // Reseta contadores
            counters = { papel: 0, plastico: 0, vidro: 0, metal: 0 };
            document.getElementById('countPapel').innerText = '0';
            document.getElementById('countPlastico').innerText = '0';
            document.getElementById('countVidro').innerText = '0';
            document.getElementById('countMetal').innerText = '0';
            
            document.getElementById('scoreDisplay').innerText = score;
            document.getElementById('timeDisplay').innerText = timeLeft;
            player.x = width / 2;
            player.targetX = width / 2;
            
            document.body.style.backgroundColor = '#f0fdf4';
            document.body.style.backgroundImage = 'radial-gradient(#bbf7d0 1px, transparent 1px)';

            document.getElementById('mainMenu').classList.add('hidden');
            document.getElementById('gameOverScreen').classList.add('hidden');
            document.getElementById('hud').classList.remove('hidden');

            // Inicia o fluxo de contagem regressiva em vez de jogar direto
            gameState = 'countdown';
            updateAudioState();
            
            startCountdownFlow();
            
            if (!animationId) {
                animationId = requestAnimationFrame(gameLoop);
            }
        }

        function startCountdownFlow() {
            const countdownScreen = document.getElementById('countdownScreen');
            const countdownText = document.getElementById('countdownText');
            countdownScreen.classList.remove('hidden');
            
            playCrowdCheer();
            
            let count = 3;
            countdownText.innerText = count;
            playCountdownBeep(false);

            const countdownInterval = setInterval(() => {
                count--;
                if (count > 0) {
                    countdownText.innerText = count;
                    playCountdownBeep(false);
                } else if (count === 0) {
                    countdownText.innerText = "VAI!";
                    playCountdownBeep(true);
                } else {
                    clearInterval(countdownInterval);
                    countdownScreen.classList.add('hidden');
                    
                    // Transição real para o gameplay
                    gameState = 'playing';
                    updateAudioState();
                    lastTime = performance.now(); // Reseta o timer para evitar bugs de delta
                }
            }, 1000);
        }

        async function endGame() {
            gameState = 'gameover';
            updateAudioState();
            playGameOverSound();
            
            document.getElementById('hud').classList.add('hidden');
            document.getElementById('gameOverScreen').classList.remove('hidden');
            document.getElementById('gameOverScreen').classList.add('flex');
            document.getElementById('finalScoreDisplay').innerText = score;
            
            const statusDiv = document.getElementById('saveStatus');
            
            if (isFirebaseActive && user) {
                statusDiv.innerText = "Salvando pontuação no ranking...";
                statusDiv.className = "text-sm text-blue-600 font-bold mb-6 h-6 animate-pulse";
                try {
                    const leaderboardRef = collection(db, 'artifacts', appId, 'public', 'data', 'a3p_ranking');
                    await addDoc(leaderboardRef, {
                        email: playerEmail,
                        score: score,
                        timestamp: Date.now()
                    });
                    statusDiv.innerText = "Pontuação salva com sucesso!";
                    statusDiv.className = "text-sm text-green-600 font-bold mb-6 h-6";
                } catch (error) {
                    console.error("Erro ao salvar:", error);
                    statusDiv.innerText = "Erro ao salvar no servidor.";
                    statusDiv.className = "text-sm text-red-600 font-bold mb-6 h-6";
                }
            } else {
                statusDiv.innerText = "Ranking online offline no momento.";
                statusDiv.className = "text-sm text-gray-500 font-bold mb-6 h-6";
            }
        }

        // --- Leaderboard UI ---
        
        function maskEmail(email) {
            const parts = email.split('@');
            if (parts.length !== 2) return email;
            const name = parts[0];
            const domain = parts[1];
            if (name.length <= 3) return name[0] + "***@" + domain;
            return name.substring(0, 3) + "***@" + domain;
        }

        async function showRanking() {
            document.getElementById('leaderboardScreen').classList.remove('hidden');
            const listContainer = document.getElementById('rankingList');
            const loading = document.getElementById('rankingLoading');
            
            listContainer.innerHTML = '';
            listContainer.classList.add('hidden');
            loading.classList.remove('hidden');

            if (!isFirebaseActive || !user) {
                loading.innerText = "Serviço de ranking indisponível.";
                return;
            }

            try {
                const leaderboardRef = collection(db, 'artifacts', appId, 'public', 'data', 'a3p_ranking');
                const querySnapshot = await getDocs(leaderboardRef);
                
                let scores = [];
                querySnapshot.forEach((doc) => {
                    scores.push(doc.data());
                });

                scores.sort((a, b) => b.score - a.score);
                scores = scores.slice(0, 50);

                if (scores.length === 0) {
                    loading.innerText = "Nenhuma pontuação registrada ainda. Seja o primeiro!";
                    return;
                }

                loading.classList.add('hidden');
                listContainer.classList.remove('hidden');

                scores.forEach((s, index) => {
                    const item = document.createElement('div');
                    item.className = `flex justify-between items-center p-3 rounded-xl border ${index === 0 ? 'bg-yellow-50 border-yellow-300' : index === 1 ? 'bg-gray-50 border-gray-300' : index === 2 ? 'bg-orange-50 border-orange-300' : 'bg-white border-gray-100'}`;
                    
                    let medal = `${index + 1}º`;
                    if (index === 0) medal = '🥇';
                    if (index === 1) medal = '🥈';
                    if (index === 2) medal = '🥉';

                    item.innerHTML = `
                        <div class="flex items-center gap-3">
                            <span class="font-black text-lg w-8 text-center text-gray-500">${medal}</span>
                            <span class="font-semibold text-gray-700 truncate max-w-[150px] sm:max-w-[200px]" title="${s.email}">${maskEmail(s.email)}</span>
                        </div>
                        <div class="font-black text-green-600 text-xl">${s.score} <span class="text-xs text-gray-400">pts</span></div>
                    `;
                    listContainer.appendChild(item);
                });

            } catch (error) {
                console.error("Erro ao carregar ranking:", error);
                loading.innerText = "Erro ao carregar o ranking.";
            }
        }

        // --- Gemini API Integration ---
        const apiKey = "";
        
        async function fetchGeminiResponse(prompt) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: "Você é um especialista em sustentabilidade e educação ambiental do programa A3P da Secretaria de Fazenda (Sefaz). Seja sempre breve (máximo de 2 frases curtas), simpático e use emojis. Foque no descarte correto no ambiente de trabalho corporativo." }] }
            };

            const delays = [1000, 2000, 4000, 8000, 16000];
            let attempt = 0;

            while (attempt <= 5) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    const data = await response.json();
                    return data.candidates?.[0]?.content?.parts?.[0]?.text || "Ops, não consegui processar a resposta agora.";
                } catch (error) {
                    if (attempt === 5) {
                        return "Desculpe, os servidores da IA estão sobrecarregados no momento. Tente novamente mais tarde!";
                    }
                    await new Promise(res => setTimeout(res, delays[attempt]));
                    attempt++;
                }
            }
        }
